210double RlsProgram::fitness(vector<int> &genome)
211{
212 double value=0.0;
213 vector<double> dx;
214 dx.resize(dimension);
215 Data xx1;
216 xx1.resize(dimension);
217 Data xx2;
218 xx2.resize(dimension);
219 int ok_flag=0;
220 vector<double> lmargin,rmargin;
221 lmargin=problem->getLeftMargin();
222 rmargin=problem->getRightMargin();
223 int ok_flag1=0;
224 int ok_flag2=0;
225 int redo=0;
226 string totalstr=printRandomProgram(genome,redo);
227 if(redo>=REDO_MAX) {
228 return
229 -1e+100;
230 }
231 int lastpos=0;
232 for(int i=0;i<dimension;i++)
233 {
234 string str="";
235 for(int j=lastpos;j<totalstr.size() && totalstr[j]!='#';j++)
236 {
237 str=str+totalstr[j];
238 lastpos++;
239 }
240 lastpos++;
241 int icode=parser.Parse(str,"x");
242 if(icode!=-1)
243 {
244 return -1e+100;
245 }
246
247 dx[i]=parser.Eval(&x0[i]);
248 if(isnan(dx[i]) || isinf(dx[i])|| parser.EvalError())
249 {
250 return -1e+100;
251 dx[i]=0.0;
252 ok_flag++;
253 ok_flag1++;
254 ok_flag2++;
255 }
256
257 xx1[i]=x0[i]+dx[i];
258 int pflag=0;
259 if(xx1[i]>rmargin[i] || xx1[i]<lmargin[i])
260 {
261 ok_flag++;
262 xx1[i]=x0[i];
263 ok_flag1++;
264 }
265 xx2[i]=x0[i]-dx[i];
266 if(xx2[i]>rmargin[i] || xx2[i]<lmargin[i])
267 {
268 ok_flag++;
269 xx2[i]=x0[i];
270 ok_flag2++;
271 }
272
273 }
274 if(ok_flag>=dimension) return -fvalue;// {return -1e+100;}
275 double val1,val2;
276 if(problem->isPointIn(xx1) && ok_flag1<dimension)
277 val1=problem->funmin(xx1);
278 else
279 {
280 val1=fvalue+1;
281 }
282 if(val1<fvalue)
283 {
284 if(isnan(val1) || isinf(val1)) return -1e+100;
285 if(val1<bestValue)
286 {
287 for(int i=0;i<dimension;i++) bestx0[i]=xx1[i];
288 bestValue=val1;
289 }
290 return -val1;
291 }
292 else
293 {
294 if(problem->isPointIn(xx2) && ok_flag2<dimension)
295 val2=problem->funmin(xx2);
296 else
297 val2=fvalue+1;
298 if(val2<fvalue)
299 {
300 if(isnan(val2) || isinf(val2)) return -1e+100;
301 if(val2<bestValue)
302 {
303 for(int i=0;i<dimension;i++) bestx0[i]=xx2[i];
304 bestValue=val2;
305 }
306 return -val2;
307 }
308 else
309 return -fvalue;
310 }
311 return 0.0;
312}
