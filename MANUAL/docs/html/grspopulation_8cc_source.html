133void GrsPopulation::mutate()
134{
135 int start = elitism * (int)(genome_count*selection_rate);
136 start = elitism;
137 for(int i=start;i<genome_count;i++)
138 {
139 for(int j=0;j<genome_size;j++)
140 {
